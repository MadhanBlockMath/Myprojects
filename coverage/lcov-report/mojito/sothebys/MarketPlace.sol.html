/**
 * @title IWrapperNativeToken
 * @dev Interface for Wrapped native tokens such as WETH, WMATIC, WBNB, etc
 */
interface IWrappedNativeToken {
    function deposit() external payable;
 
    function withdraw(uint256 wad) external;
 
    function transfer(address to, uint256 value) external returns (bool);
}
 
contract Marketplace is ReentrancyGuard, ERC1155TokenReceiver {
    using SafeMath for uint256;
    using SafeERC20 for IERC20;
    using ECDSA for bytes32;
 
    struct Order {
        string uuid;
        uint256 tokenId;
        address tokenContract;
        uint256 quantity;
        address payable signer; 
        uint256 fixedPrice;
        address paymentToken;
        address whitelistedBuyer;
    }
 
    // Interface ID constants
    bytes4 constant ERC721_INTERFACE_ID = 0x80ac58cd;
    bytes4 constant ERC1155_INTERFACE_ID = 0xd9b67a26;
    bytes4 constant ROYALTIES_CREATORCORE_INTERFACE_ID = 0xbb3bafd6;
 
    // The address of the NFT protocol to use via this contract
    address public nftAddress;
 
    // ERC-20 adress of the Native token (can be WETH, WBNB, WMATIC, etc)
    IWrappedNativeToken public nativeToken;
 
    // Address of the Marketplace
    address payable public marketplaceAddress;
 
    // Fee percentage to the Platform
    uint256 public platformFeePercentage;
 
    // UUID validation on orders
    mapping(string => bool) public usedUUID;
 
    event BuyExecuted(

    constructor(
        address _nftAddress,
        address _nativeTokenAddress,
        address _marketplaceAddress,
        uint256 _platformFeePercentage
    ) {
        require(
            (IERC165(_nftAddress).supportsInterface(ERC721_INTERFACE_ID) ||
                IERC165(_nftAddress).supportsInterface(ERC1155_INTERFACE_ID)),
            "Doesn't support NFT interface"
        );
 
        nftAddress = _nftAddress;
        nativeToken = IWrappedNativeToken(_nativeTokenAddress);
        marketplaceAddress = payable(_marketplaceAddress);
        platformFeePercentage = _platformFeePercentage;
    }
 
    function buy(Order memory order, bytes memory sellerSignature)
        external
        payable
        nonReentrant
    {
        require(
            (IERC165(order.tokenContract).supportsInterface(
                ERC721_INTERFACE_ID
            ) ||
                IERC165(order.tokenContract).supportsInterface(
                    ERC1155_INTERFACE_ID
                )),
            "tokenContract does not support ERC721 or ERC1155 interface"
        );
 
        require(
            order.whitelistedBuyer == address(0) ||
                order.whitelistedBuyer == msg.sender,
            "msg.sender should be the whitelisted buyer"
        );
 
        // Validating signatures
        require(
            _verifySignature(order, sellerSignature, order.signer),
            "Invalid seller signature"
        );
 
        // Validating UUID
        require(!usedUUID[order.uuid], "UUID already used");
 
        // Updaiting the Used UUID
        usedUUID[order.uuid] = true;
 
        _tokenTransaction (order, order.signer, msg.sender);
 
        uint256 tokenOwnerProfit = order.fixedPrice;
 
        require(
            order.paymentToken != address(0) || msg.value == order.fixedPrice,
            "Sent ETH Value does not match specified buy amount"
        );
 
        uint256 platformFee;
        if (marketplaceAddress != address(0) && platformFeePercentage > 0) {
            platformFee = tokenOwnerProfit.mul(platformFeePercentage).div(100);
            tokenOwnerProfit = tokenOwnerProfit.sub(platformFee);
 
            _handlePayment(msg.sender, marketplaceAddress, order.paymentToken, platformFee);
        }
        tokenOwnerProfit = _handleRoyaltyEnginePayout(
            order.tokenContract,
            msg.sender,
            order.tokenId,
            tokenOwnerProfit,
            order.paymentToken
        );
        _handlePayment(msg.sender, order.signer, order.paymentToken, tokenOwnerProfit);
 
        emit BuyExecuted(
            order.uuid,
            order.tokenId,
            order.tokenContract,
            order.quantity,
            order.signer,
            msg.sender,
            order.fixedPrice,
            order.paymentToken,
            marketplaceAddress,
            platformFee
        );
    }
 
    function sell(Order memory order, bytes memory buyerSignature) 
        external
        payable
        nonReentrant
    {
        require(
            (IERC165(order.tokenContract).supportsInterface(
                ERC721_INTERFACE_ID
            ) ||
                IERC165(order.tokenContract).supportsInterface(
                    ERC1155_INTERFACE_ID
                )),
            "tokenContract does not support ERC721 or ERC1155 interface"
        );
 
        require(
            _verifySignature(order, buyerSignature, order.signer),
            "Invalid Buyer signature"
        );
 
        // Validating UUID
        require(!usedUUID[order.uuid], "UUID already used");
 
        // Updaiting the Used UUID
        usedUUID[order.uuid] = true;
 
        _tokenTransaction(order, msg.sender, order.signer);
 
        uint256 tokenOwnerProfit = order.fixedPrice;
 
        uint256 platformFee;
        if (marketplaceAddress != address(0) && platformFeePercentage > 0) {
            platformFee = tokenOwnerProfit.mul(platformFeePercentage).div(100);
            tokenOwnerProfit = tokenOwnerProfit.sub(platformFee);
 
            _handlePayment(order.signer, marketplaceAddress, order.paymentToken, platformFee);
        }  
 
        tokenOwnerProfit = _handleRoyaltyEnginePayout(
            order.tokenContract,
            order.signer,
            order.tokenId,
            tokenOwnerProfit,
            order.paymentToken
        );  
 
        _handlePayment(order.signer, payable(msg.sender), order.paymentToken, tokenOwnerProfit);
 
        emit SellExecuted(
            order.uuid, 
            order.tokenId, 
            order.tokenContract, 
            order.quantity, 
            msg.sender, 
            order.signer, 
            order.fixedPrice, 
            order.paymentToken, 
            marketplaceAddress, 
            platformFee
        );
    }
 
    function _handlePayment(
        address from,
        address payable to,
        address paymentToken,
        uint256 amount
    ) internal {
        if (paymentToken != address(0)) {
 
            IERC20(paymentToken).safeTransferFrom(from, to, amount);
 
        } else {
            (bool success, ) = to.call{value: amount}("");
            if (!success) {
 
                IERC20(address(nativeToken)).safeTransferFrom(from, to, amount);
            }
        }
    }
 
    function _tokenTransaction(Order memory order, address tokenOwner, address buyer) internal {
 
        if (
            IERC165(order.tokenContract).supportsInterface(ERC721_INTERFACE_ID)
        ) {
            require(
                IERC721(order.tokenContract).ownerOf(order.tokenId) ==
                    tokenOwner,
                "maker is not the owner"
            );
 
            // Transferring the ERC721
            IERC721(order.tokenContract).safeTransferFrom(
                tokenOwner,
                buyer,
                order.tokenId
            );
        } else {
            uint256 ownerBalance = IERC1155(order.tokenContract).balanceOf(
                tokenOwner,
                order.tokenId
            );
            require(
                order.quantity <= ownerBalance && order.quantity > 0,
                "Insufficeint token balance"
            );
 
            // Transferring the ERC1155
            IERC1155(order.tokenContract).safeTransferFrom(
                tokenOwner,
                buyer,
                order.tokenId,
                order.quantity,
                "0x"
            );
        }
    }
 
    function _handleRoyaltyEnginePayout(
        address _tokenContract,
        address _from,
        uint256 _tokenId,
        uint256 _amount,
        address _payoutCurrency
    ) internal returns (uint256) {
        // Store the initial amount
        uint256 amountRemaining = _amount;
 
        // Verifying whether the token contract supports Royalties of Creator Core
        if (
            IERC165(_tokenContract).supportsInterface(
                ROYALTIES_CREATORCORE_INTERFACE_ID
            )
        ) {
            // Get the royalty recipients and their associated amounts
            (
                address payable[] memory recipients,
                uint256[] memory amountBasis
            ) = ICreatorCore(_tokenContract).getRoyalties(_tokenId);
 
            // Store the number of recipients
            uint256 totalRecipients = recipients.length;
 
            // If there are no royalties, return the initial amount
            if (totalRecipients == 0) return _amount;
 
            // Payout each royalty
            for (uint256 i = 0; i < totalRecipients; ) {
                // Cache the recipient and amount
                address payable recipient = recipients[i];
                uint256 amount = (amountBasis[i] * _amount) / 10000;
 
                // Ensure that we aren't somehow paying out more than we have
                require(amountRemaining >= amount, "insolvent");
 
                _handlePayment(_from, recipient, _payoutCurrency, amount);
                emit RoyaltyPayout(_tokenContract, _tokenId, recipient, amount);
 
                // Cannot underflow as remaining amount is ensured to be greater than or equal to royalty amount
                unchecked {
                    amountRemaining -= amount;
                    ++i;
                }
            }
        }
 
        return amountRemaining;
    }
 
    function _verifySignature(
        Order memory _order,
        bytes memory _signature,
        address _signer
    ) internal pure returns (bool) {
        return
            keccak256(
                abi.encodePacked(
                    _order.uuid,
                    _order.tokenId,
                    _order.tokenContract,
                    _order.quantity,
                    _order.signer,
                    _order.fixedPrice,
                    _order.paymentToken
                    
                )
            ).toEthSignedMessageHash().recover(_signature) == _signer;
    }

                    _order.fixedPrice,
                    _order.paymentToken
                )
            );
    }
